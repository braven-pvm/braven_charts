### File: `src/commands/verify.ts`

```typescript
import { Command } from 'commander';
import chalk from 'chalk';
import { 
  loadConfig, 
  loadManifest, 
  loadProgress,
  loadVerificationYaml,
  runVerificationChecks,
  VerificationResult,
  CheckResult
} from '../core';

interface VerifyOptions {
  json?: boolean;
  verbose?: boolean;
  task?: number;
}

export function registerVerifyCommand(program: Command): void {
  program
    .command('verify')
    .description('Run verification checks for current/specified task')
    .option('--task <number>', 'Task number to verify (default: previous task)')
    .option('--json', 'Output as JSON')
    .option('-v, --verbose', 'Show detailed check output')
    .action(async (options: VerifyOptions) => {
      try {
        const result = await verify(options);
        
        if (options.json) {
          console.log(JSON.stringify(result, null, 2));
        } else {
          printVerificationResult(result, options.verbose ?? false);
        }
        
        process.exit(result.passed ? 0 : 3);
      } catch (error) {
        console.error(chalk.red('Error:'), error.message);
        process.exit(1);
      }
    });
}

async function verify(options: VerifyOptions): Promise<VerificationResult> {
  const config = await loadConfig();
  const manifest = await loadManifest(config);
  const progress = await loadProgress(config);
  
  // Determine which task to verify
  const taskNum = options.task ?? progress.current_task - 1;
  const task = manifest.tasks.find(t => t.id === taskNum);
  
  if (!task) {
    throw new Error(`Task ${taskNum} not found in manifest`);
  }
  
  // Load verification YAML
  const verificationPath = `${config.paths.verification}/task-${taskNum.toString().padStart(3, '0')}.yaml`;
  const verification = await loadVerificationYaml(verificationPath);
  
  // Run all checks
  const result = await runVerificationChecks(verification, {
    manifestTask: task,
    specPath: config.paths.spec,
    orchestraRoot: config.orchestraRoot,
  });
  
  return result;
}

function printVerificationResult(result: VerificationResult, verbose: boolean): void {
  console.log();
  console.log(chalk.bold(`Task ${result.taskId} Verification`));
  console.log('─'.repeat(40));
  
  // Summary
  const passedChecks = result.checks.filter(c => c.passed).length;
  const totalChecks = result.checks.length;
  
  if (result.passed) {
    console.log(chalk.green.bold(`✓ PASSED (${passedChecks}/${totalChecks} checks)`));
  } else {
    console.log(chalk.red.bold(`✗ FAILED (${passedChecks}/${totalChecks} checks)`));
  }
  
  console.log();
  
  // By severity
  const bySeverity = {
    blocking: result.checks.filter(c => c.severity === 'blocking'),
    major: result.checks.filter(c => c.severity === 'major'),
    minor: result.checks.filter(c => c.severity === 'minor'),
    info: result.checks.filter(c => c.severity === 'info'),
  };
  
  for (const [severity, checks] of Object.entries(bySeverity)) {
    if (checks.length === 0) continue;
    
    const passed = checks.filter(c => c.passed).length;
    const icon = passed === checks.length ? '✓' : '✗';
    const color = passed === checks.length ? chalk.green : 
                  severity === 'blocking' ? chalk.red : chalk.yellow;
    
    console.log(color(`${icon} ${severity.toUpperCase()}: ${passed}/${checks.length}`));
    
    if (verbose || !checks.every(c => c.passed)) {
      for (const check of checks) {
        const checkIcon = check.passed ? chalk.green('✓') : chalk.red('✗');
        console.log(`  ${checkIcon} ${check.id}: ${check.description}`);
        if (!check.passed && check.message) {
          console.log(`      ${chalk.dim(check.message)}`);
        }
      }
    }
  }
  
  // Decision
  console.log();
  const blockingFailed = bySeverity.blocking.some(c => !c.passed);
  const majorFailed = bySeverity.major.filter(c => !c.passed).length >= 2;
  
  if (blockingFailed) {
    console.log(chalk.red('Decision: FAIL - Blocking checks failed'));
    console.log(chalk.dim('All blocking checks must pass for verification to succeed.'));
  } else if (majorFailed) {
    console.log(chalk.red('Decision: FAIL - 2+ major checks failed'));
    console.log(chalk.dim('Maximum 1 major failure allowed.'));
  } else {
    console.log(chalk.green('Decision: PASS'));
  }
}
```

### File: `src/core/verification.ts`

```typescript
import fs from 'fs/promises';
import path from 'path';
import { execSync } from 'child_process';
import yaml from 'js-yaml';
import { z } from 'zod';

// Schema for verification YAML
const VerificationCheckSchema = z.object({
  id: z.string(),
  description: z.string(),
  type: z.enum(['file_exists', 'file_contains', 'command', 'test_count', 'custom']),
  severity: z.enum(['blocking', 'major', 'minor', 'info']),
  params: z.record(z.any()).optional(),
});

const VerificationYamlSchema = z.object({
  task_ref: z.string(),
  blocking_checks: z.array(VerificationCheckSchema).optional(),
  major_checks: z.array(VerificationCheckSchema).optional(),
  minor_checks: z.array(VerificationCheckSchema).optional(),
  info_checks: z.array(VerificationCheckSchema).optional(),
  screenshot: z.object({
    required: z.boolean(),
    path: z.string().optional(),
    verify: z.array(z.string()).optional(),
  }).optional(),
});

export type VerificationYaml = z.infer<typeof VerificationYamlSchema>;
export type VerificationCheck = z.infer<typeof VerificationCheckSchema>;

export interface CheckResult {
  id: string;
  description: string;
  severity: 'blocking' | 'major' | 'minor' | 'info';
  passed: boolean;
  message?: string;
}

export interface VerificationResult {
  taskId: number;
  passed: boolean;
  checks: CheckResult[];
  timestamp: string;
}

export async function loadVerificationYaml(filePath: string): Promise<VerificationYaml> {
  const content = await fs.readFile(filePath, 'utf-8');
  const parsed = yaml.load(content);
  return VerificationYamlSchema.parse(parsed);
}

export async function runVerificationChecks(
  verification: VerificationYaml,
  context: {
    manifestTask: any;
    specPath: string;
    orchestraRoot: string;
  }
): Promise<VerificationResult> {
  const allChecks: VerificationCheck[] = [
    ...(verification.blocking_checks ?? []).map(c => ({ ...c, severity: 'blocking' as const })),
    ...(verification.major_checks ?? []).map(c => ({ ...c, severity: 'major' as const })),
    ...(verification.minor_checks ?? []).map(c => ({ ...c, severity: 'minor' as const })),
    ...(verification.info_checks ?? []).map(c => ({ ...c, severity: 'info' as const })),
  ];
  
  const results: CheckResult[] = [];
  
  for (const check of allChecks) {
    const result = await runSingleCheck(check, context);
    results.push(result);
  }
  
  // Determine overall pass/fail
  const blockingFailed = results.some(r => r.severity === 'blocking' && !r.passed);
  const majorFailures = results.filter(r => r.severity === 'major' && !r.passed).length;
  const passed = !blockingFailed && majorFailures < 2;
  
  return {
    taskId: context.manifestTask.id,
    passed,
    checks: results,
    timestamp: new Date().toISOString(),
  };
}

async function runSingleCheck(
  check: VerificationCheck,
  context: any
): Promise<CheckResult> {
  try {
    switch (check.type) {
      case 'file_exists':
        return await checkFileExists(check, context);
      case 'file_contains':
        return await checkFileContains(check, context);
      case 'command':
        return await checkCommand(check, context);
      case 'test_count':
        return await checkTestCount(check, context);
      case 'custom':
        return await checkCustom(check, context);
      default:
        return {
          id: check.id,
          description: check.description,
          severity: check.severity,
          passed: false,
          message: `Unknown check type: ${check.type}`,
        };
    }
  } catch (error) {
    return {
      id: check.id,
      description: check.description,
      severity: check.severity,
      passed: false,
      message: error.message,
    };
  }
}

async function checkFileExists(check: VerificationCheck, context: any): Promise<CheckResult> {
  const filePath = check.params?.path;
  try {
    await fs.access(filePath);
    return {
      id: check.id,
      description: check.description,
      severity: check.severity,
      passed: true,
    };
  } catch {
    return {
      id: check.id,
      description: check.description,
      severity: check.severity,
      passed: false,
      message: `File not found: ${filePath}`,
    };
  }
}

async function checkFileContains(check: VerificationCheck, context: any): Promise<CheckResult> {
  const { path: filePath, pattern } = check.params ?? {};
  const content = await fs.readFile(filePath, 'utf-8');
  const regex = new RegExp(pattern);
  const passed = regex.test(content);
  
  return {
    id: check.id,
    description: check.description,
    severity: check.severity,
    passed,
    message: passed ? undefined : `Pattern not found: ${pattern}`,
  };
}

async function checkCommand(check: VerificationCheck, context: any): Promise<CheckResult> {
  const { command, expected_exit } = check.params ?? {};
  try {
    execSync(command, { stdio: 'pipe' });
    return {
      id: check.id,
      description: check.description,
      severity: check.severity,
      passed: expected_exit === 0,
    };
  } catch (error) {
    return {
      id: check.id,
      description: check.description,
      severity: check.severity,
      passed: expected_exit !== 0,
      message: error.message,
    };
  }
}

async function checkTestCount(check: VerificationCheck, context: any): Promise<CheckResult> {
  const { test_dir, min_count } = check.params ?? {};
  // Run flutter test and count
  try {
    const output = execSync(`flutter test ${test_dir} --reporter compact`, { 
      encoding: 'utf-8',
      stdio: ['pipe', 'pipe', 'pipe']
    });
    const match = output.match(/(\d+) tests passed/);
    const count = match ? parseInt(match[1]) : 0;
    const passed = count >= min_count;
    
    return {
      id: check.id,
      description: check.description,
      severity: check.severity,
      passed,
      message: passed ? `${count} tests passed` : `Only ${count} tests (need ${min_count})`,
    };
  } catch (error) {
    return {
      id: check.id,
      description: check.description,
      severity: check.severity,
      passed: false,
      message: `Test run failed: ${error.message}`,
    };
  }
}

async function checkCustom(check: VerificationCheck, context: any): Promise<CheckResult> {
  // Custom checks would be plugin-based in full implementation
  return {
    id: check.id,
    description: check.description,
    severity: check.severity,
    passed: true,
    message: 'Custom check not implemented',
  };
}
```
