## Overview

Initialize the TypeScript project structure for the Orchestra CLI with all required dependencies, tooling configuration, and initial scaffolding.

## Objectives

1. Create `tools/orchestra-cli/` directory structure
2. Initialize npm package with TypeScript configuration
3. Configure build tooling (tsx for development, tsup for production)
4. Set up testing framework (Vitest)
5. Create CLI entry point with Commander.js

## Prerequisites

- Node.js 18+ installed
- npm or pnpm available
- Repository cloned locally

### 7. tests/setup.ts

```typescript
import { beforeEach, afterEach, vi } from 'vitest';
import * as fs from 'node:fs';
import * as path from 'node:path';
import * as os from 'node:os';

// Global test fixtures
let tempDir: string;

beforeEach(() => {
  // Create temp directory for each test
  tempDir = fs.mkdtempSync(path.join(os.tmpdir(), 'orchestra-test-'));
  
  // Suppress console output during tests
  vi.spyOn(console, 'log').mockImplementation(() => {});
  vi.spyOn(console, 'error').mockImplementation(() => {});
});

afterEach(() => {
  // Clean up temp directory
  if (tempDir && fs.existsSync(tempDir)) {
    fs.rmSync(tempDir, { recursive: true, force: true });
  }
  
  // Restore mocks
  vi.restoreAllMocks();
});

// Export helpers
export function getTempDir(): string {
  return tempDir;
}

export function createTempFile(filename: string, content: string): string {
  const filePath = path.join(tempDir, filename);
  fs.mkdirSync(path.dirname(filePath), { recursive: true });
  fs.writeFileSync(filePath, content, 'utf-8');
  return filePath;
}

export function createTempDir(dirname: string): string {
  const dirPath = path.join(tempDir, dirname);
  fs.mkdirSync(dirPath, { recursive: true });
  return dirPath;
}
```
