## Overview

Implement `orchestra complete` - marks task as verified, records results, and advances to next task.

## Dependencies

- [x] Task 1.1: Project setup
- [x] Task 1.2: Core libraries

## Inputs

| Input | Source | Purpose |
|-------|--------|---------|
| Commit message | CLI argument | Git commit message |
| Manifest | `.orchestra/orchestrator/.orchestrator-only/manifest.yaml` | Task list |
| Progress | `.orchestra/orchestrator/.orchestrator-only/progress.yaml` | Current state |

| Output | Location | Purpose |
|--------|----------|---------|

| Task results | `.orchestra/orchestrator/results/task-NNN-results.md` | Verification record |
| Git commit | Repository | Commit with message |

### File: `src/commands/complete.ts`

```typescript
import { Command } from 'commander';
import chalk from 'chalk';
import { 
  loadConfig, 
  loadManifest, 
  loadProgress,
  updateProgress,
  createTaskResults,
  commitChanges,
  getGitStatus,
  ProgressTask
} from '../core';

interface CompleteOptions {
  message: string;
  json?: boolean;
  noCommit?: boolean;
  force?: boolean;
}

interface CompleteResult {
  completedTask: number;
  nextTask: number | null;
  committed: boolean;
  commitHash?: string;
  message: string;
}

export function registerCompleteCommand(program: Command): void {
  program
    .command('complete')
    .description('Complete current task, record results, advance to next')
    .requiredOption('-m, --message <message>', 'Commit message')
    .option('--json', 'Output as JSON')
    .option('--no-commit', 'Skip git commit')
    .option('--force', 'Complete even if verification not run')
    .action(async (options: CompleteOptions) => {
      try {
        const result = await complete(options);
        
        if (options.json) {
          console.log(JSON.stringify(result, null, 2));
        } else {
          printCompleteResult(result);
        }
        
        process.exit(0);
      } catch (error) {
        console.error(chalk.red('Error:'), error.message);
        process.exit(1);
      }
    });
}

async function complete(options: CompleteOptions): Promise<CompleteResult> {
  const config = await loadConfig();
  const manifest = await loadManifest(config);
  const progress = await loadProgress(config);
  
  const currentTaskNum = progress.current_task;
  const currentTask = manifest.tasks.find(t => t.id === currentTaskNum);
  
  if (!currentTask) {
    throw new Error(`Current task ${currentTaskNum} not found in manifest`);
  }
  
  // Check verification was run (unless --force)
  const taskProgress = progress.tasks?.[currentTaskNum];
  if (!options.force && taskProgress?.status !== 'verified') {
    throw new Error(
      `Task ${currentTaskNum} has not been verified. Run 'orchestra verify' first, or use --force.`
    );
  }
  
  // Check for uncommitted changes
  const gitStatus = await getGitStatus();
  if (!options.noCommit && gitStatus.uncommitted.length === 0) {
    throw new Error('No uncommitted changes to commit. Make changes first or use --no-commit.');
  }
  
  // Create results file
  const resultsPath = await createTaskResults({
    taskId: currentTaskNum,
    taskTitle: currentTask.title,
    status: 'passed',
    message: options.message,
    timestamp: new Date().toISOString(),
  }, config);
  
  // Update progress
  const nextTaskNum = currentTaskNum < manifest.tasks.length ? currentTaskNum + 1 : null;
  
  const updatedProgress = {
    ...progress,
    current_task: nextTaskNum ?? currentTaskNum,
    tasks: {
      ...progress.tasks,
      [currentTaskNum]: {
        ...taskProgress,
        status: 'completed',
        completed_at: new Date().toISOString(),
        commit_message: options.message,
      } as ProgressTask,
      ...(nextTaskNum && {
        [nextTaskNum]: {
          status: 'pending',
          attempts: 0,
        } as ProgressTask,
      }),
    },
  };
  
  await updateProgress(updatedProgress, config);
  
  // Git commit
  let commitHash: string | undefined;
  if (!options.noCommit) {
    const prefix = config.git?.commit_prefix ?? 'orchestra';
    const fullMessage = `${prefix}(Task ${currentTaskNum}): ${options.message}`;
    
    commitHash = await commitChanges({
      message: fullMessage,
      files: [
        config.paths.progress,
        resultsPath,
        // Also stage any implementor changes
        '.',
      ],
    });
  }
  
  return {
    completedTask: currentTaskNum,
    nextTask: nextTaskNum,
    committed: !options.noCommit,
    commitHash,
    message: options.message,
  };
}

function printCompleteResult(result: CompleteResult): void {
  console.log();
  console.log(chalk.green.bold(`✓ Task ${result.completedTask} Completed`));
  console.log('─'.repeat(40));
  
  console.log(`Message: ${result.message}`);
  
  if (result.committed) {
    console.log(`Commit: ${chalk.cyan(result.commitHash?.slice(0, 7))}`);
  } else {
    console.log(`Commit: ${chalk.dim('skipped')}`);
  }
  
  if (result.nextTask) {
    console.log();
    console.log(`Next: Task ${result.nextTask}`);
    console.log(chalk.dim(`Run 'orchestra prepare --task ${result.nextTask}' to continue.`));
  } else {
    console.log();
    console.log(chalk.green.bold('🎉 Sprint Complete!'));
    console.log('All tasks finished. Great work!');
  }
}
```

```typescript
import fs from 'fs/promises';
import path from 'path';
import { Config } from './config';

export interface TaskResultsInput {
  taskId: number;
  taskTitle: string;
  status: 'passed' | 'failed' | 'skipped';
  message: string;
  timestamp: string;
  verificationResult?: any;
}

export async function createTaskResults(
  input: TaskResultsInput,
  config: Config
): Promise<string> {
  const resultsDir = path.join(config.orchestraRoot, 'orchestrator/results');
  await fs.mkdir(resultsDir, { recursive: true });
  
  const fileName = `task-${input.taskId.toString().padStart(3, '0')}-results.md`;
  const filePath = path.join(resultsDir, fileName);
  
  const content = `# Task ${input.taskId} Results

## Summary

| Field | Value |
|-------|-------|
| Task | ${input.taskId}: ${input.taskTitle} |
| Status | ${input.status.toUpperCase()} |
| Completed | ${input.timestamp} |
| Message | ${input.message} |

## Verification

${input.verificationResult ? formatVerificationResult(input.verificationResult) : 'Verification details not recorded.'}

## Notes

Task completed via \`orchestra complete\`.
`;

  await fs.writeFile(filePath, content, 'utf-8');
  return filePath;
}

function formatVerificationResult(result: any): string {
  if (!result.checks) return 'No check details available.';
  
  const lines = ['| Check | Severity | Result |', '|-------|----------|--------|'];
  for (const check of result.checks) {
    const status = check.passed ? '✓ Pass' : '✗ Fail';
    lines.push(`| ${check.id} | ${check.severity} | ${status} |`);
  }
  return lines.join('\n');
}
```

```typescript
import simpleGit, { SimpleGit } from 'simple-git';

let git: SimpleGit | null = null;

function getGit(): SimpleGit {
  if (!git) {
    git = simpleGit();
  }
  return git;
}

export interface GitStatus {
  branch: string;
  uncommitted: string[];
  staged: string[];
  untracked: string[];
}

export async function getGitStatus(): Promise<GitStatus> {
  const g = getGit();
  const status = await g.status();
  
  return {
    branch: status.current ?? 'unknown',
    uncommitted: [...status.modified, ...status.created, ...status.deleted],
    staged: status.staged,
    untracked: status.not_added,
  };
}

export interface CommitOptions {
  message: string;
  files: string[];
}

export async function commitChanges(options: CommitOptions): Promise<string> {
  const g = getGit();
  
  // Stage files
  await g.add(options.files);
  
  // Commit
  const result = await g.commit(options.message);
  
  return result.commit;
}

export async function getCurrentBranch(): Promise<string> {
  const g = getGit();
  const status = await g.status();
  return status.current ?? 'unknown';
}
```

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
```

## Acceptance Criteria

- [ ] Fails if task not verified (unless --force)
- [ ] Creates task results file in results directory
- [ ] Updates progress.yaml: marks current task complete, sets next task
- [ ] Commits changes with prefixed message
- [ ] Returns next task number (or null if sprint complete)
- [ ] --no-commit skips git commit
- [ ] --force bypasses verification check
- [ ] JSON output includes all result details
- [ ] Human output shows completion summary and next steps

## Notes

- Verification check ensures tasks can't be skipped
- Results file serves as audit trail
- Next task auto-advance streamlines workflow
- Git commit includes both orchestra state changes and implementor work
