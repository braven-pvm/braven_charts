# Task 1.5: Prepare Command

## Overview

Implement the `orchestra prepare` command that prepares the handover folder for a specific task. Generates `current-task.md` from templates, updates context files, and readies the handover for the implementor.

## Objectives

1. Create the prepare command with all options
2. Implement task validation (dependencies, status)
3. Generate handover documents from templates
4. Update manifest with task status
5. Clear previous verification artifacts

## Prerequisites

- Task 1.1 (Project Setup)
- Task 1.2 (Core Libraries)
- Task 1.4 (Init Command) - for templates

### 1. src/commands/prepare.ts

```typescript
import { Command } from 'commander';
import * as fs from 'node:fs';
import * as path from 'node:path';
import chalk from 'chalk';
import ora from 'ora';
import {
  loadConfig,
  loadManifest,
  saveManifest,
  getTask,
  getNextTask,
  updateTaskStatus,
  getOrchestraPath,
  ManifestError,
  TaskError,
} from '../lib/index.js';
import { renderTemplate } from '../lib/templates.js';
import * as output from '../lib/output.js';
import type { Manifest, Task } from '../models/index.js';

interface PrepareOptions {
  task?: string;
  force?: boolean;
  dryRun?: boolean;
  json?: boolean;
}

export function createPrepareCommand(): Command {
  return new Command('prepare')
    .description('Prepare handover for next task')
    .option('--task <id>', 'Specific task ID to prepare')
    .option('-f, --force', 'Prepare even if previous task incomplete')
    .option('--dry-run', 'Show what would be generated')
    .option('--json', 'Output as JSON')
    .action(async (options: PrepareOptions) => {
      await runPrepare(options);
    });
}

async function runPrepare(options: PrepareOptions): Promise<void> {
  const spinner = options.json ? null : ora('Preparing task...').start();

  try {
    const manifest = loadManifest();
    
    // Find task to prepare
    let task: Task | undefined;
    
    if (options.task) {
      task = getTask(manifest, options.task);
      if (!task) {
        throw new TaskError(`Task not found: ${options.task}`);
      }
    } else {
      task = getNextTask(manifest);
      if (!task) {
        throw new ManifestError('No pending tasks available');
      }
    }

    // Validate task can be prepared
    validatePrepare(manifest, task, options.force ?? false);

    if (options.dryRun) {
      spinner?.stop();
      showDryRun(task);
      return;
    }

    // Generate handover documents
    const handoverPath = getOrchestraPath('handover');
    
    // Clear verification folder
    const verificationPath = path.join(handoverPath, 'verification');
    if (fs.existsSync(verificationPath)) {
      fs.rmSync(verificationPath, { recursive: true });
    }
    fs.mkdirSync(verificationPath, { recursive: true });

    // Generate current-task.md
    const currentTaskContent = renderTemplate('current-task.md', {
      id: task.id,
      title: task.title,
      description: task.description ?? '',
      deliverables: task.acceptance_criteria ?? [],
      dependencies: getTaskDependencies(manifest, task),
    });
    fs.writeFileSync(
      path.join(handoverPath, 'current-task.md'),
      currentTaskContent
    );

    // Generate completion-signal.md template
    const signalContent = renderTemplate('completion-signal.md', {
      id: task.id,
      title: task.title,
      deliverables: task.acceptance_criteria ?? [],
    });
    fs.writeFileSync(
      path.join(handoverPath, 'completion-signal.md'),
      signalContent
    );

    // Generate task-context.md
    const contextContent = renderTemplate('task-context.md', {
      sprint: manifest.metadata,
      currentTask: task,
      stats: getTaskStats(manifest),
      completedTasks: manifest.tasks.filter(t => t.status === 'completed'),
    });
    fs.writeFileSync(
      path.join(handoverPath, 'task-context.md'),
      contextContent
    );

    // Update manifest - mark task as in-progress
    const updatedManifest = updateTaskStatus(manifest, task.id, 'in-progress');
    saveManifest(updatedManifest);

    spinner?.succeed('Task prepared successfully');

    if (options.json) {
      console.log(JSON.stringify({
        success: true,
        task: {
          id: task.id,
          title: task.title,
          status: 'in-progress',
        },
        files: [
          'handover/current-task.md',
          'handover/completion-signal.md',
          'handover/task-context.md',
        ],
      }));
    } else {
      showSuccess(task);
    }
  } catch (error) {
    spinner?.fail('Failed to prepare task');
    
    if (options.json) {
      console.log(JSON.stringify({
        success: false,
        error: String(error),
      }));
    } else {
      output.error(String(error));
    }
    process.exit(1);
  }
}

function validatePrepare(manifest: Manifest, task: Task, force: boolean): void {
  // Check task status
  if (task.status !== 'not-started') {
    throw new TaskError(
      `Task ${task.id} is already ${task.status}`,
      { taskId: task.id, status: task.status }
    );
  }

  // Check for in-progress task
  if (!force) {
    const inProgress = manifest.tasks.find(t => t.status === 'in-progress');
    if (inProgress) {
      throw new TaskError(
        `Task ${inProgress.id} is still in progress. Complete it first or use --force.`,
        { blockingTask: inProgress.id }
      );
    }
  }

  // Check dependencies
  const deps = task.depends_on ?? [];
  for (const depId of deps) {
    const dep = getTask(manifest, depId);
    if (!dep) {
      throw new TaskError(`Dependency not found: ${depId}`, { taskId: task.id, dependency: depId });
    }
    if (dep.status !== 'completed') {
      throw new TaskError(
        `Dependency ${depId} is not complete (${dep.status})`,
        { taskId: task.id, dependency: depId, dependencyStatus: dep.status }
      );
    }
  }
}

function getTaskDependencies(manifest: Manifest, task: Task): Array<{ id: string; title: string; completed: boolean }> {
  const deps = task.depends_on ?? [];
  return deps.map(depId => {
    const dep = getTask(manifest, depId);
    return {
      id: depId,
      title: dep?.title ?? 'Unknown',
      completed: dep?.status === 'completed',
    };
  });
}

function getTaskStats(manifest: Manifest) {
  return {
    completed: manifest.tasks.filter(t => t.status === 'completed').length,
    inProgress: manifest.tasks.filter(t => t.status === 'in-progress').length,
    notStarted: manifest.tasks.filter(t => t.status === 'not-started').length,
    total: manifest.tasks.length,
  };
}

function showDryRun(task: Task): void {
  output.header('Dry Run - Would Create:');
  
  console.log(chalk.bold('\nTask:'));
  console.log(`  ${task.id}: ${task.title}`);
  
  console.log(chalk.bold('\nFiles to generate:'));
  console.log(`  ${chalk.green('+')} handover/current-task.md`);
  console.log(`  ${chalk.green('+')} handover/completion-signal.md`);
  console.log(`  ${chalk.green('+')} handover/task-context.md`);
  
  console.log(chalk.bold('\nManifest update:'));
  console.log(`  ${task.id}: not-started → ${chalk.blue('in-progress')}`);
}

function showSuccess(task: Task): void {
  console.log('');
  output.success('Task prepared!');
  console.log('');
  console.log(chalk.bold('Task:') + ` ${task.id} - ${task.title}`);
  console.log('');
  console.log(chalk.bold('Handover files:'));
  console.log('  ✓ handover/current-task.md');
  console.log('  ✓ handover/completion-signal.md');
  console.log('  ✓ handover/task-context.md');
  console.log('  ✓ handover/verification/ (cleared)');
  console.log('');
  console.log(chalk.bold('Next:') + ' Implementor can begin work');
}

export { runPrepare };
```

### 2. tests/commands/prepare.test.ts

```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import * as fs from 'node:fs';
import * as path from 'node:path';
import { runPrepare } from '../../src/commands/prepare.js';
import { createTempFile, createTempDir, getTempDir } from '../setup.js';

describe('prepare command', () => {
  beforeEach(() => {
    vi.spyOn(process, 'cwd').mockReturnValue(getTempDir());
    
    // Set up initialized orchestra
    createTempDir('.orchestra');
    createTempFile('.orchestra/orchestra.yaml', 'version: "1.0.0"\norchestra_dir: ".orchestra"');
    createTempDir('.orchestra/handover');
    createTempDir('.orchestra/common/templates');
    
    // Create templates
    createTempFile('.orchestra/common/templates/current-task.md.hbs', 
      '# Task {{id}}: {{title}}\n{{description}}');
    createTempFile('.orchestra/common/templates/completion-signal.md.hbs',
      '# Signal for {{id}}');
    createTempFile('.orchestra/common/templates/task-context.md.hbs',
      '# Context');
  });

  it('should prepare next pending task', async () => {
    createTempFile('.orchestra/manifest.yaml', `
version: "1.0.0"
metadata:
  feature_id: "TEST-001"
  title: "Test Feature"
tasks:
  - id: "TASK-001"
    title: "First Task"
    status: "not-started"
`);

    await runPrepare({});
    
    // Check handover files created
    const handoverDir = path.join(getTempDir(), '.orchestra', 'handover');
    expect(fs.existsSync(path.join(handoverDir, 'current-task.md'))).toBe(true);
    expect(fs.existsSync(path.join(handoverDir, 'completion-signal.md'))).toBe(true);
  });

  it('should prepare specific task by ID', async () => {
    createTempFile('.orchestra/manifest.yaml', `
version: "1.0.0"
metadata:
  feature_id: "TEST-001"
  title: "Test Feature"
tasks:
  - id: "TASK-001"
    title: "First Task"
    status: "not-started"
  - id: "TASK-002"
    title: "Second Task"
    status: "not-started"
`);

    await runPrepare({ task: 'TASK-002' });
    
    const currentTask = fs.readFileSync(
      path.join(getTempDir(), '.orchestra', 'handover', 'current-task.md'),
      'utf-8'
    );
    expect(currentTask).toContain('TASK-002');
  });

  it('should fail if task has incomplete dependencies', async () => {
    createTempFile('.orchestra/manifest.yaml', `
version: "1.0.0"
metadata:
  feature_id: "TEST-001"
  title: "Test Feature"
tasks:
  - id: "TASK-001"
    title: "First Task"
    status: "not-started"
  - id: "TASK-002"
    title: "Second Task"
    status: "not-started"
    depends_on:
      - "TASK-001"
`);

    const exitSpy = vi.spyOn(process, 'exit').mockImplementation(() => {
      throw new Error('exit');
    });

    await expect(runPrepare({ task: 'TASK-002' })).rejects.toThrow('exit');
    expect(exitSpy).toHaveBeenCalledWith(1);
  });

  it('should update manifest status to in-progress', async () => {
    createTempFile('.orchestra/manifest.yaml', `
version: "1.0.0"
metadata:
  feature_id: "TEST-001"
  title: "Test Feature"
tasks:
  - id: "TASK-001"
    title: "First Task"
    status: "not-started"
`);

    await runPrepare({});
    
    const manifest = fs.readFileSync(
      path.join(getTempDir(), '.orchestra', 'manifest.yaml'),
      'utf-8'
    );
    expect(manifest).toContain('in-progress');
  });

  it('should show dry run without modifying files', async () => {
    createTempFile('.orchestra/manifest.yaml', `
version: "1.0.0"
metadata:
  feature_id: "TEST-001"
  title: "Test Feature"
tasks:
  - id: "TASK-001"
    title: "First Task"
    status: "not-started"
`);

    await runPrepare({ dryRun: true });
    
    // Manifest should not be modified
    const manifest = fs.readFileSync(
      path.join(getTempDir(), '.orchestra', 'manifest.yaml'),
      'utf-8'
    );
    expect(manifest).not.toContain('in-progress');
  });
});
```
