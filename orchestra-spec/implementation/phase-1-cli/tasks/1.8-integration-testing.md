## Implementation

### Test Helpers (`test/helpers/setup.ts`)

```typescript
import { mkdtemp, rm, cp, mkdir, writeFile, readFile } from 'fs/promises';
import { tmpdir } from 'os';
import { join } from 'path';
import { execSync } from 'child_process';

export interface TestWorkspace {
  root: string;
  orchestraDir: string;
  manifestPath: string;
  cleanup: () => Promise<void>;
}

/**
 * Creates an isolated test workspace with git initialized
 */
export async function createTestWorkspace(
  options: {
    withGit?: boolean;
    withManifest?: boolean;
    manifestContent?: string;
    withOrchestraDir?: boolean;
  } = {}
): Promise<TestWorkspace> {
  const {
    withGit = true,
    withManifest = false,
    manifestContent,
    withOrchestraDir = false,
  } = options;

  // Create temp directory
  const root = await mkdtemp(join(tmpdir(), 'orchestra-test-'));
  const orchestraDir = join(root, '.orchestra');
  const manifestPath = join(orchestraDir, 'manifest.yaml');

  // Initialize git if requested
  if (withGit) {
    execSync('git init', { cwd: root, stdio: 'pipe' });
    execSync('git config user.email "test@test.com"', { cwd: root, stdio: 'pipe' });
    execSync('git config user.name "Test User"', { cwd: root, stdio: 'pipe' });
    
    // Create initial commit
    await writeFile(join(root, 'README.md'), '# Test Project');
    execSync('git add .', { cwd: root, stdio: 'pipe' });
    execSync('git commit -m "Initial commit"', { cwd: root, stdio: 'pipe' });
  }

  // Create .orchestra directory if requested
  if (withOrchestraDir) {
    await mkdir(orchestraDir, { recursive: true });
    await mkdir(join(orchestraDir, 'templates'), { recursive: true });
    await mkdir(join(orchestraDir, 'docs'), { recursive: true });
  }

  // Create manifest if requested
  if (withManifest && manifestContent) {
    await mkdir(orchestraDir, { recursive: true });
    await writeFile(manifestPath, manifestContent);
  }

  return {
    root,
    orchestraDir,
    manifestPath,
    cleanup: async () => {
      await rm(root, { recursive: true, force: true });
    },
  };
}

/**
 * Creates a git branch in the test workspace
 */
export async function createBranch(
  workspace: TestWorkspace,
  branchName: string,
  checkout = true
): Promise<void> {
  const cmd = checkout
    ? `git checkout -b ${branchName}`
    : `git branch ${branchName}`;
  execSync(cmd, { cwd: workspace.root, stdio: 'pipe' });
}

/**
 * Makes a commit in the test workspace
 */
export async function makeCommit(
  workspace: TestWorkspace,
  message: string,
  files: Record<string, string> = {}
): Promise<string> {
  // Create/modify files
  for (const [path, content] of Object.entries(files)) {
    const fullPath = join(workspace.root, path);
    await mkdir(join(fullPath, '..'), { recursive: true });
    await writeFile(fullPath, content);
  }

  // If no files provided, create a dummy change
  if (Object.keys(files).length === 0) {
    await writeFile(
      join(workspace.root, `change-${Date.now()}.txt`),
      'test change'
    );
  }

  execSync('git add .', { cwd: workspace.root, stdio: 'pipe' });
  execSync(`git commit -m "${message}"`, { cwd: workspace.root, stdio: 'pipe' });

  // Return commit hash
  return execSync('git rev-parse HEAD', { cwd: workspace.root, stdio: 'pipe' })
    .toString()
    .trim();
}

/**
 * Gets current branch name
 */
export function getCurrentBranch(workspace: TestWorkspace): string {
  return execSync('git branch --show-current', {
    cwd: workspace.root,
    stdio: 'pipe',
  })
    .toString()
    .trim();
}

/**
 * Reads file content from workspace
 */
export async function readWorkspaceFile(
  workspace: TestWorkspace,
  relativePath: string
): Promise<string> {
  return readFile(join(workspace.root, relativePath), 'utf-8');
}

/**
 * Checks if file exists in workspace
 */
export async function fileExists(
  workspace: TestWorkspace,
  relativePath: string
): Promise<boolean> {
  try {
    await readFile(join(workspace.root, relativePath));
    return true;
  } catch {
    return false;
  }
}
```

```typescript
import { execSync } from 'child_process';
import type { TestWorkspace } from './setup';

/**
 * Gets list of commits on current branch
 */
export function getCommitLog(workspace: TestWorkspace): string[] {
  const output = execSync('git log --oneline', {
    cwd: workspace.root,
    stdio: 'pipe',
  }).toString();
  return output.trim().split('\n').filter(Boolean);
}

/**
 * Gets list of tags
 */
export function getTags(workspace: TestWorkspace): string[] {
  try {
    const output = execSync('git tag', {
      cwd: workspace.root,
      stdio: 'pipe',
    }).toString();
    return output.trim().split('\n').filter(Boolean);
  } catch {
    return [];
  }
}

/**
 * Checks if working directory is clean
 */
export function isWorkingDirClean(workspace: TestWorkspace): boolean {
  const output = execSync('git status --porcelain', {
    cwd: workspace.root,
    stdio: 'pipe',
  }).toString();
  return output.trim() === '';
}

/**
 * Gets current HEAD commit hash
 */
export function getHeadCommit(workspace: TestWorkspace): string {
  return execSync('git rev-parse HEAD', {
    cwd: workspace.root,
    stdio: 'pipe',
  })
    .toString()
    .trim();

/**
 * Checks if branch exists

export function branchExists(workspace: TestWorkspace, name: string): boolean {
```

### Custom Assertions (`test/helpers/assertions.ts`)

```typescript
import { expect } from 'vitest';
import { readFile, access } from 'fs/promises';
import { join } from 'path';
import type { TestWorkspace } from './setup';

/**
 * Assert file exists with optional content check
 */
export async function assertFileExists(
  workspace: TestWorkspace,
  relativePath: string,
  contentPattern?: RegExp
): Promise<void> {
  const fullPath = join(workspace.root, relativePath);
  
  // Check existence
  await expect(access(fullPath)).resolves.not.toThrow();
  
  // Check content if pattern provided
  if (contentPattern) {
    const content = await readFile(fullPath, 'utf-8');
    expect(content).toMatch(contentPattern);
  }
}

/**
 * Assert file does not exist
 */
export async function assertFileNotExists(
  workspace: TestWorkspace,
  relativePath: string
): Promise<void> {
  const fullPath = join(workspace.root, relativePath);
  await expect(access(fullPath)).rejects.toThrow();
}

/**
 * Assert directory structure matches expected
 */
export async function assertDirectoryStructure(
  workspace: TestWorkspace,
  basePath: string,
  expectedFiles: string[]
): Promise<void> {
  for (const file of expectedFiles) {
    await assertFileExists(workspace, join(basePath, file));
  }
}

/**
 * Assert YAML file contains expected keys
 */
export async function assertYamlContains(
  workspace: TestWorkspace,
  relativePath: string,
  expectedContent: Record<string, unknown>
): Promise<void> {
  const fullPath = join(workspace.root, relativePath);
  const content = await readFile(fullPath, 'utf-8');
  const yaml = await import('yaml');
  const parsed = yaml.parse(content);
  
  for (const [key, value] of Object.entries(expectedContent)) {
    expect(parsed).toHaveProperty(key);
    if (value !== undefined) {
      expect(parsed[key]).toEqual(value);
    }
  }
}
```

## Integration Test Suites

### Full Workflow Test (`test/integration/workflow.test.ts`)

```typescript
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import {
  createTestWorkspace,
  createBranch,
  makeCommit,
  readWorkspaceFile,
  fileExists,
  type TestWorkspace,
} from '../helpers/setup';
import { getTags, branchExists, isWorkingDirClean } from '../helpers/git';
import { assertFileExists, assertYamlContains } from '../helpers/assertions';

// Import CLI commands (through programmatic API)
import { initCommand } from '../../src/commands/init';
import { prepareCommand } from '../../src/commands/prepare';
import { verifyCommand } from '../../src/commands/verify';
import { completeCommand } from '../../src/commands/complete';
import { statusCommand } from '../../src/commands/status';

describe('Orchestra CLI: Full Workflow', () => {
  let workspace: TestWorkspace;

  beforeEach(async () => {
    workspace = await createTestWorkspace({ withGit: true });
  });

  afterEach(async () => {
    await workspace.cleanup();
  });

  describe('Complete Sprint Lifecycle', () => {
    it('should execute full workflow: init → prepare → verify → complete', async () => {
      // Step 1: Initialize Orchestra
      const initResult = await initCommand.execute({
        projectRoot: workspace.root,
        projectName: 'integration-test',
        interactive: false,
      });

      expect(initResult.success).toBe(true);
      await assertFileExists(workspace, '.orchestra/manifest.yaml');
      await assertFileExists(workspace, '.orchestra/templates/feature-plan.md.hbs');

      // Step 2: Add a sprint to manifest (simulate user editing)
      const manifestContent = `
version: "1.0"
project:
  name: integration-test
  description: Test project

sprints:
  - id: "001"
    name: Test Sprint
    description: Testing the workflow
    status: not-started
    tasks:
      - id: "001.1"
        name: First Task
        status: not-started
`;
      await writeFile(workspace.manifestPath, manifestContent);
      await makeCommit(workspace, 'Add sprint to manifest', {
        '.orchestra/manifest.yaml': manifestContent,
      });

      // Step 3: Prepare the sprint
      const prepareResult = await prepareCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
        createBranch: true,
      });

      expect(prepareResult.success).toBe(true);
      expect(branchExists(workspace, 'sprint/001-test-sprint')).toBe(true);
      await assertFileExists(workspace, '.orchestra/docs/sprints/001/readme.md');
      await assertFileExists(workspace, '.orchestra/docs/sprints/001/tasks/001.1.md');

      // Step 4: Simulate work on the sprint
      await makeCommit(workspace, 'Implement first task', {
        'src/feature.ts': 'export const feature = () => {};',
      });

      // Update task status in manifest
      const updatedManifest = manifestContent.replace(
        'status: not-started',
        'status: completed'
      );
      await makeCommit(workspace, 'Complete task 001.1', {
        '.orchestra/manifest.yaml': updatedManifest,
      });

      // Step 5: Verify the sprint
      const verifyResult = await verifyCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
      });

      expect(verifyResult.success).toBe(true);
      expect(verifyResult.checks.every((c) => c.passed)).toBe(true);

      // Step 6: Complete the sprint
      const completeResult = await completeCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
        createTag: true,
        generateReport: true,
      });

      expect(completeResult.success).toBe(true);
      expect(getTags(workspace)).toContain('sprint-001');
      await assertFileExists(
        workspace,
        '.orchestra/docs/sprints/001/completion-report.md'
      );

      // Step 7: Verify final status
      const statusResult = await statusCommand.execute({
        projectRoot: workspace.root,
      });

      expect(statusResult.currentSprint).toBeNull(); // No active sprint
      expect(statusResult.completedSprints).toContain('001');
    });

    it('should handle multiple sprints in sequence', async () => {
      // Initialize with complex manifest
      await initCommand.execute({
        projectRoot: workspace.root,
        projectName: 'multi-sprint-test',
        interactive: false,
      });

      const complexManifest = await readFile(
        'test/fixtures/manifests/complex.yaml',
        'utf-8'
      );
      await writeFile(workspace.manifestPath, complexManifest);
      await makeCommit(workspace, 'Add complex manifest');

      // Complete sprint 001 (already marked completed in fixture)
      const complete001 = await completeCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
      });
      expect(complete001.success).toBe(true);

      // Prepare sprint 002
      const prepare002 = await prepareCommand.execute({
        projectRoot: workspace.root,
        sprintId: '002',
      });
      expect(prepare002.success).toBe(true);

      // Status should show 002 as current
      const status = await statusCommand.execute({
        projectRoot: workspace.root,
      });
      expect(status.currentSprint?.id).toBe('002');
      expect(status.completedSprints).toContain('001');
    });
  });

  describe('Error Recovery Scenarios', () => {
    it('should prevent completing sprint with incomplete tasks', async () => {
      await initCommand.execute({
        projectRoot: workspace.root,
        projectName: 'error-test',
        interactive: false,
      });

      const manifest = `
version: "1.0"
project:
  name: error-test
sprints:
  - id: "001"
    name: Test
    status: in-progress
    tasks:
      - id: "001.1"
        name: Incomplete Task
        status: in-progress
`;
      await writeFile(workspace.manifestPath, manifest);

      const verifyResult = await verifyCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
      });

      expect(verifyResult.success).toBe(false);
      expect(verifyResult.checks.some((c) => !c.passed)).toBe(true);

      const completeResult = await completeCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
      });

      expect(completeResult.success).toBe(false);
      expect(completeResult.error).toMatch(/incomplete tasks/i);
    });

    it('should handle dirty working directory gracefully', async () => {
      await initCommand.execute({
        projectRoot: workspace.root,
        projectName: 'dirty-test',
        interactive: false,
      });

      // Create uncommitted changes
      await writeFile(join(workspace.root, 'uncommitted.txt'), 'dirty');

      expect(isWorkingDirClean(workspace)).toBe(false);

      const prepareResult = await prepareCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
      });

      // Should warn but allow proceeding or fail gracefully
      // (depends on implementation choice)
      expect(prepareResult.warnings).toContain(
        expect.stringMatching(/uncommitted changes/i)
      );
    });
  });

  describe('Idempotency Tests', () => {
    it('should handle re-running init safely', async () => {
      // First init
      await initCommand.execute({
        projectRoot: workspace.root,
        projectName: 'idempotent-test',
        interactive: false,
      });

      // Capture state
      const manifest1 = await readWorkspaceFile(
        workspace,
        '.orchestra/manifest.yaml'
      );

      // Second init (should not overwrite)
      const result = await initCommand.execute({
        projectRoot: workspace.root,
        interactive: false,
      });

      expect(result.success).toBe(true);
      expect(result.alreadyInitialized).toBe(true);

      // Manifest should be unchanged
      const manifest2 = await readWorkspaceFile(
        workspace,
        '.orchestra/manifest.yaml'
      );
      expect(manifest2).toBe(manifest1);
    });

    it('should handle re-running prepare for same sprint', async () => {
      await initCommand.execute({
        projectRoot: workspace.root,
        projectName: 'prepare-test',
        interactive: false,
      });

      const manifest = `
version: "1.0"
project:
  name: prepare-test
sprints:
  - id: "001"
    name: Test
    status: not-started
    tasks:
      - id: "001.1"
        name: Task
        status: not-started
`;
      await writeFile(workspace.manifestPath, manifest);
      await makeCommit(workspace, 'Add manifest');

      // First prepare
      await prepareCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
      });

      // Second prepare (should be idempotent)
      const result = await prepareCommand.execute({
        projectRoot: workspace.root,
        sprintId: '001',
      });

      expect(result.success).toBe(true);
      expect(result.alreadyPrepared).toBe(true);
    });
  });
});
```

### Status Command Integration Test (`test/integration/status.test.ts`)

```typescript
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { createTestWorkspace, type TestWorkspace } from '../helpers/setup';
import { statusCommand } from '../../src/commands/status';
import { initCommand } from '../../src/commands/init';
import { writeFile } from 'fs/promises';

describe('Orchestra CLI: Status Command Integration', () => {
  let workspace: TestWorkspace;

  beforeEach(async () => {
    workspace = await createTestWorkspace({ withGit: true });
  });

  afterEach(async () => {
    await workspace.cleanup();
  });

  it('should report uninitialized state', async () => {
    const result = await statusCommand.execute({
      projectRoot: workspace.root,
    });

    expect(result.initialized).toBe(false);
    expect(result.currentSprint).toBeNull();
  });

  it('should report initialized but no sprints', async () => {
    await initCommand.execute({
      projectRoot: workspace.root,
      projectName: 'status-test',
      interactive: false,
    });

    const result = await statusCommand.execute({
      projectRoot: workspace.root,
    });

    expect(result.initialized).toBe(true);
    expect(result.currentSprint).toBeNull();
    expect(result.totalSprints).toBe(0);
  });

  it('should report current sprint details', async () => {
    await initCommand.execute({
      projectRoot: workspace.root,
      projectName: 'status-test',
      interactive: false,
    });

    const manifest = `
version: "1.0"
project:
  name: status-test
sprints:
  - id: "001"
    name: Active Sprint
    status: in-progress
    tasks:
      - id: "001.1"
        name: Done Task
        status: completed
      - id: "001.2"
        name: Doing Task
        status: in-progress
      - id: "001.3"
        name: Todo Task
        status: not-started
`;
    await writeFile(workspace.manifestPath, manifest);

    const result = await statusCommand.execute({
      projectRoot: workspace.root,
    });

    expect(result.currentSprint).toBeDefined();
    expect(result.currentSprint?.id).toBe('001');
    expect(result.currentSprint?.name).toBe('Active Sprint');
    expect(result.currentSprint?.taskProgress).toEqual({
      completed: 1,
      inProgress: 1,
      notStarted: 1,
      total: 3,
    });
  });

  it('should report sprint history', async () => {
    await initCommand.execute({
      projectRoot: workspace.root,
      projectName: 'history-test',
      interactive: false,
    });

    const manifest = `
version: "1.0"
project:
  name: history-test
sprints:
  - id: "001"
    name: First Sprint
    status: completed
    completedAt: "2024-01-15"
  - id: "002"
    name: Second Sprint  
    status: completed
    completedAt: "2024-02-01"
  - id: "003"
    name: Current Sprint
    status: in-progress
`;
    await writeFile(workspace.manifestPath, manifest);

    const result = await statusCommand.execute({
      projectRoot: workspace.root,
      includeHistory: true,
    });

    expect(result.completedSprints).toHaveLength(2);
    expect(result.completedSprints).toContain('001');
    expect(result.completedSprints).toContain('002');
    expect(result.currentSprint?.id).toBe('003');
  });
});
```
