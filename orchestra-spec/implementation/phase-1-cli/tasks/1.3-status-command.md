## Overview

Implement the `orchestra status` command that displays the current state of a sprint, including progress, current task, phase information, and activity history.

## Objectives

1. Create the status command with all options
2. Implement default, brief, and detailed views
3. Add task and phase detail views
4. Implement metrics calculation
5. Support JSON and human-readable output

### 1. src/commands/status.ts

```typescript
import { Command } from 'commander';
import chalk from 'chalk';
import {
  loadConfig,
  loadManifest,
  loadProgress,
  getTask,
  getCurrentTask,
  getTaskStats,
  isOrchestraInitialized,
} from '../lib/index.js';
import * as output from '../lib/output.js';
import type { Manifest, Task, ProgressLog } from '../models/index.js';

interface StatusOptions {
  task?: string;
  phase?: number;
  history?: boolean;
  metrics?: boolean;
  json?: boolean;
  brief?: boolean;
}

export function createStatusCommand(): Command {
  return new Command('status')
    .description('Show current Orchestra status')
    .option('--task <id>', 'Show details for specific task')
    .option('--phase <number>', 'Show details for specific phase', parseInt)
    .option('--history', 'Include task history')
    .option('--metrics', 'Include sprint metrics')
    .option('--json', 'Output as JSON')
    .option('--brief', 'One-line summary only')
    .action(async (options: StatusOptions) => {
      await runStatus(options);
    });
}

async function runStatus(options: StatusOptions): Promise<void> {
  // Check initialization
  if (!isOrchestraInitialized()) {
    output.error('Orchestra not initialized. Run "orchestra init" first.');
    process.exit(1);
  }

  try {
    const manifest = loadManifest();
    const progress = loadProgress(manifest.metadata.feature_id);

    // Handle specific views
    if (options.task) {
      await showTaskDetail(manifest, options.task, options.json);
      return;
    }

    if (options.phase !== undefined) {
      await showPhaseDetail(manifest, options.phase, options.json);
      return;
    }

    if (options.brief) {
      showBriefStatus(manifest);
      return;
    }

    // Default view
    await showFullStatus(manifest, progress, options);
  } catch (error) {
    if (options.json) {
      console.log(JSON.stringify({ error: String(error) }));
    } else {
      output.error(String(error));
    }
    process.exit(1);
  }
}

function showBriefStatus(manifest: Manifest): void {
  const stats = getTaskStats(manifest);
  const current = getCurrentTask(manifest);
  const currentId = current?.id ?? 'none';
  const percent = Math.round((stats.completed / stats.total) * 100);
  
  console.log(
    `Orchestra: ${manifest.metadata.feature_id} | ` +
    `Task ${currentId}/${stats.total} | ` +
    `${percent}% complete`
  );
}

async function showFullStatus(
  manifest: Manifest,
  progress: ProgressLog,
  options: StatusOptions
): Promise<void> {
  const stats = getTaskStats(manifest);
  const current = getCurrentTask(manifest);

  if (options.json) {
    const jsonOutput = {
      sprint: {
        id: manifest.metadata.feature_id,
        title: manifest.metadata.title,
        status: current ? 'in_progress' : 'idle',
      },
      progress: {
        total: stats.total,
        completed: stats.completed,
        inProgress: stats.inProgress,
        notStarted: stats.notStarted,
        failed: stats.failed,
        percentage: Math.round((stats.completed / stats.total) * 100),
      },
      currentTask: current ? {
        id: current.id,
        title: current.title,
        status: current.status,
        attempt: current.attempt_count ?? 1,
      } : null,
      ...(options.metrics ? { metrics: calculateMetrics(manifest, progress) } : {}),
      ...(options.history ? { history: getRecentTasks(manifest, 10) } : {}),
    };
    console.log(JSON.stringify(jsonOutput, null, 2));
    return;
  }

  // Header
  output.header('Orchestra Status');
  output.divider('═', 60);

  // Sprint info
  console.log(`\n${chalk.bold('Sprint:')} ${manifest.metadata.feature_id}`);
  console.log(`${chalk.bold('Title:')}  ${manifest.metadata.title}`);
  
  // Progress bar
  console.log('\n' + chalk.bold('Progress'));
  output.divider('─', 40);
  console.log(output.formatManifestOverview(manifest));

  // Current task
  if (current) {
    console.log('\n' + chalk.bold('Current Task'));
    output.divider('─', 40);
    console.log(output.formatTask(current, true));
  } else {
    console.log('\n' + chalk.dim('No task in progress'));
  }

  // Recent activity
  const recentTasks = getRecentTasks(manifest, 3);
  if (recentTasks.length > 0) {
    console.log('\n' + chalk.bold('Recent Activity'));
    output.divider('─', 40);
    for (const task of recentTasks) {
      console.log(output.formatTask(task));
    }
  }

  // Metrics (if requested)
  if (options.metrics) {
    console.log('\n' + chalk.bold('Metrics'));
    output.divider('─', 40);
    const metrics = calculateMetrics(manifest, progress);
    console.log(output.formatKeyValue({
      'First-attempt pass rate': `${metrics.firstAttemptPassRate}%`,
      'Average attempts': metrics.averageAttempts.toFixed(2),
      'Completed tasks': metrics.completedTasks,
    }));
  }

  // History (if requested)
  if (options.history) {
    console.log('\n' + chalk.bold('Full History'));
    output.divider('─', 40);
    for (const task of manifest.tasks) {
      console.log(output.formatTask(task));
    }
  }
}

async function showTaskDetail(
  manifest: Manifest,
  taskId: string,
  json?: boolean
): Promise<void> {
  const task = getTask(manifest, taskId);
  
  if (!task) {
    if (json) {
      console.log(JSON.stringify({ error: `Task not found: ${taskId}` }));
    } else {
      output.error(`Task not found: ${taskId}`);
    }
    process.exit(2);
  }

  if (json) {
    console.log(JSON.stringify(task, null, 2));
    return;
  }

  output.header('Task Details');
  output.divider('═', 60);
  
  console.log(`\n${chalk.bold('Task:')} ${task.id}`);
  console.log(`${chalk.bold('Title:')} ${task.title}`);
  console.log(`${chalk.bold('Status:')} ${output.formatStatus(task.status)}`);
  
  if (task.description) {
    console.log(`\n${chalk.bold('Description:')}`);
    console.log(chalk.dim(task.description));
  }

  if (task.depends_on && task.depends_on.length > 0) {
    console.log(`\n${chalk.bold('Dependencies:')}`);
    for (const depId of task.depends_on) {
      const dep = getTask(manifest, depId);
      const depStatus = dep ? output.formatStatus(dep.status) : chalk.dim('unknown');
      console.log(`  ${depStatus} ${depId}: ${dep?.title ?? 'Unknown'}`);
    }
  }

  if (task.acceptance_criteria && task.acceptance_criteria.length > 0) {
    console.log(`\n${chalk.bold('Acceptance Criteria:')}`);
    for (const criterion of task.acceptance_criteria) {
      console.log(`  • ${criterion}`);
    }
  }

  console.log(`\n${chalk.bold('Attempts:')} ${task.attempt_count ?? 0}/${task.max_attempts ?? 3}`);
  
  if (task.started_at) {
    console.log(`${chalk.bold('Started:')} ${task.started_at}`);
  }
  if (task.completed_at) {
    console.log(`${chalk.bold('Completed:')} ${task.completed_at}`);
  }
}

async function showPhaseDetail(
  manifest: Manifest,
  phaseNumber: number,
  json?: boolean
): Promise<void> {
  // For now, phases are implicit based on task grouping
  // Future: Add explicit phase support in manifest
  if (json) {
    console.log(JSON.stringify({ 
      phase: phaseNumber, 
      message: 'Phase details not yet implemented' 
    }));
  } else {
    output.warning(`Phase details not yet implemented (phase ${phaseNumber})`);
  }
}

function getRecentTasks(manifest: Manifest, count: number): Task[] {
  return manifest.tasks
    .filter(t => t.status === 'completed')
    .sort((a, b) => {
      const aTime = a.completed_at ? new Date(a.completed_at).getTime() : 0;
      const bTime = b.completed_at ? new Date(b.completed_at).getTime() : 0;
      return bTime - aTime;
    })
    .slice(0, count);
}

interface Metrics {
  completedTasks: number;
  firstAttemptPassRate: number;
  averageAttempts: number;
}

function calculateMetrics(manifest: Manifest, progress: ProgressLog): Metrics {
  const completed = manifest.tasks.filter(t => t.status === 'completed');
  const firstAttemptPasses = completed.filter(t => (t.attempt_count ?? 1) === 1);
  
  const totalAttempts = completed.reduce((sum, t) => sum + (t.attempt_count ?? 1), 0);
  
  return {
    completedTasks: completed.length,
    firstAttemptPassRate: completed.length > 0 
      ? Math.round((firstAttemptPasses.length / completed.length) * 100) 
      : 0,
    averageAttempts: completed.length > 0 
      ? totalAttempts / completed.length 
      : 0,
  };
}

export { runStatus };
```

import { describe, it, expect, beforeEach, vi } from 'vitest';
import { runStatus } from '../../src/commands/status.js';
import { createTempFile, getTempDir } from '../setup.js';

describe('status command', () => {
  beforeEach(() => {
    // Set up test fixtures
    vi.spyOn(process, 'cwd').mockReturnValue(getTempDir());
  });

  it('should show error when not initialized', async () => {
    const exitSpy = vi.spyOn(process, 'exit').mockImplementation(() => {
      throw new Error('exit');
    });

    expect(exitSpy).toHaveBeenCalledWith(1);

  it('should show brief status', async () => {

    createTempFile('.orchestra/manifest.yaml', `
version: "1.0.0"
metadata:
  feature_id: "TEST-001"
  title: "Test Feature"
tasks:
  - id: "TASK-001"
    title: "Test Task"
    status: "not-started"
`);

    const consoleSpy = vi.spyOn(console, 'log');
    
    await runStatus({ brief: true });
    
    expect(consoleSpy).toHaveBeenCalledWith(
      expect.stringContaining('TEST-001')
    );

    // Set up fixtures...

});

- [ ] `orchestra status` shows sprint overview with progress bar
